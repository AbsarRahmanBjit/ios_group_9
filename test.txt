struct Stack<E> {
    var  items:[E] = []
    
    mutating func push(_ item:E) {
        items.insert(item, at: 0)
    }
    
    mutating func pop() -> E {
        items.removeFirst()
    }
    
    mutating fun jony4() {

    }
    
}
func swap(_ a: inout Int,_ b: inout Int) {
    let temp = a
    a = b
    b = temp //temp will be assigned to b
}
